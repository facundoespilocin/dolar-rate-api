getProducts() {
            this.showOverlay = true;
            this.isBusy = true;

            let resource = "/products";

            let params = {
                currentPage: this.currentPage,
                pageSize: this.pageSize,
                searchBy: this.searchByFilterSelected.label,
                query: this.inputFilter,
                status: this.stateSelected,
            }

            let promise = this.axios.get(resource, { params });

            return promise.then((res) => {
                this.products = res?.data?.products;
                this.totalRows = res?.data?.totalRows;
                this.isBusy = false;
                this.showOverlay = false;
                
                if (this.inputFilter !== "") {
                    this.$refs.productsTable.refresh();
                }

                return(this.products)
            }).catch(e => {
                //console.log(e);
                this.showNotification("error", "No fue posible cargar los Productos. Error: " + e);
                return [];
            }).finally(e => {
                this.showOverlay = false;
                this.isBusy = false;
            })
        },
